704.二分查找

```js
var search = function (nums, target) {
  let res = -1;
  let i = 0,
    j = nums.length - 1;
  //这一步经常出错
  while (i <= j) {
    let k = Math.floor((i + j) / 2);
    if (target < nums[k]) {
      j = k - 1;
    } else if (target > nums[k]) {
      i = k + 1;
    } else {
      res = k;
      return res;
    }
  }
  return res;
};
```
